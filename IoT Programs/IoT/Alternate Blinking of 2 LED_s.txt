// Alternate Blinking of 2 LED's

import RPi.GPIO as m
import time
m.setmode(m.BOARD)
m.setwarnings(False)
m.setup(18,m.OUT)
m.setup(10,m.OUT)
while True:
        m.output(18,True)
        print("LED1 is ON")
        time.sleep(5)
        m.output(18,False)
        print("LED1 is OFF")
        m.output(10,True)
        print("LED2 is ON")
        time.sleep(7)
        m.output(10,False)
        print("LED2 is OFF")
