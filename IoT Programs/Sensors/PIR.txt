import RPi.GPIO as p
import time
p.setmode(m.BOARD)
p.setwarnings(False)
p.setup(11,p.IN)
p.setup(3,p.OUT)
try:
	while True:
        	i=p.input(11)
		if i==0:
			print("No Intruder")
			p.output(3,0)

       		elif i==1:
			print("Intruder Detected")
			p.output(3,1)

	p.cleanup()
